@override
Future<Color?> getIconColor({
  required AppModel appModel,
  required DictionaryHeading heading,
}) async {
  bool hasDuplicates = await appModel.checkForDuplicates(heading.term);
  if (hasDuplicates) {
    return Colors.red;
  } else {
    return null;
  }
}
